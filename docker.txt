# Create EKS Cluster
eksctl create cluster --name trend-eks-cluster --region us-east-1 --nodegroup-name trend-nodes \
--node-type t3.medium \
--nodes 2 \
--nodes-min 1 \
--nodes-max 3 \
--managed


# Connect kubectl to Cluster
